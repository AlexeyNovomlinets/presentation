WebSocket API

Create WebSocket

						WebSocket WebSocket(
  in DOMString url,
  in optional DOMString protocols
);

http:
const ws = new WebSocket('ws://echo.websocket.org');

https:
const wss = new WebSocket('wss://echo.websocket.org', 'someProtocol');

Properties

							WebSocket.binaryType;
WebSocket.readyState;
WebSocket.bufferedAmount;
WebSocket.extensions;
WebSocket.protocol;
WebSocket.url;

WebSocket.onopen = () => {};
WebSocket.onmessage = () => {};
WebSocket.onclose = () => {};
WebSocket.onerror = () => {};

Methods

							WebSocket.send(data);
WebSocket.close([code[, reason]]);

Example

							const ws = new WebSocket('ws://echo.websocket.org');

ws.onopen = () => {
  console.log('Online!');
  ws.send('String data');
};

ws.onmessage = event => {
  console.log(event.data);
  ws.close();
};

ws.onclose = event => console.log('Offline!');
ws.onerror = error => console.log('WebSocket Error: ' + error);
